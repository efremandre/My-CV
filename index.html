	function towelSort(matrix) {
		const newArr = [];
		const result = [];
	
		if (arguments.length === 0 || matrix.length === 0) {
			return result;
		} else {
			matrix.forEach((elem, index) => {
		(index % 2 === 0) ? newArr.push(elem) : newArr.push(elem.reverse());
		});
	}
	
	return result.concat(...newArr);
	}
